bool CompareImageByAcquisitionTime(const LPS_DCMImgInfo & v1,const LPS_DCMImgInfo & v2)
{
	// return 1;
	logGlobal.Error("CompareImageByAcquisitionTime begin");
	DcmDataset* pDataset1= v1->pDcmFile->getDataset();
    DcmDataset* pDataset2= v2->pDcmFile->getDataset();
	const char*  acquisitionTime1,*acquisitionTime2;
	logGlobal.Error("before pDataset1->findAndGetString(DCM_AcquisitionTime,acquisitionTime1);");
 OFCondition ofcond =	pDataset1->findAndGetString(DCM_AcquisitionTime,acquisitionTime1);
 if(ofcond.bad())
     acquisitionTime1="";
  ofcond= pDataset2->findAndGetString(DCM_AcquisitionTime,acquisitionTime2);
    if(ofcond.bad())
     acquisitionTime2="";
   {
		CString strTemp;
		strTemp.Format("acquisitionTime1=%s",acquisitionTime1);
		logGlobal.Error(strTemp.GetBuffer());
		strTemp.Format("acquisitionTime2=%s",acquisitionTime2);
		logGlobal.Error(strTemp.GetBuffer());
		strTemp.ReleaseBuffer();
	}
   logGlobal.Error("before strcmp ");
    int result= strcmp(acquisitionTime1,acquisitionTime2);
      if(result>0)
		  return false;
	  else
         return true ;
}
